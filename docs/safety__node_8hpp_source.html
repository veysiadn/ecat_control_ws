69template<typename FutureT, typename WaitTimeT>
70std::future_status
71wait_for_result(
72 FutureT & future,
73 WaitTimeT time_to_wait)
74{
75 auto end = std::chrono::steady_clock::now() + time_to_wait;
76 std::chrono::milliseconds wait_period(100);
77 std::future_status status = std::future_status::timeout;
78 do {
79 auto now = std::chrono::steady_clock::now();
80 auto time_left = end - now;
81 if (time_left <= std::chrono::milliseconds(0)) {break;}
82 status = future.wait_for((time_left < wait_period) ? time_left : wait_period);
83 } while (status != std::future_status::ready);
84 return status;
85}
